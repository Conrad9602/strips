(define (problem build-barracks)
  (:domain starcraft)
  (:init (and (scv scv) (location sector-a) (location sector-b)
         (location mineral-field-a) (location mineral-field-b) 
         (minerals mineral-field-a) (minerals mineral-field-b) 
         (at scv sector-a)))
  (:goal (and (barracks sector-a)))
)
